<script>
  // Typewriter
  const text = "Code như hacker, sống như samurai.";
  let i = 0;
  function type() { if (i < text.length) { document.getElementById("typewriter").innerHTML += text.charAt(i); i++; setTimeout(type, 80); } }
  setTimeout(type, 1000);

  // Slogan Live
  const slogans = ["Code như hacker, sống như samurai.", "Việt Nam #1", "Fullstack Cyber Ninja", "Dẫn đầu tương lai!"];
  let s = 0;
  setInterval(() => { document.getElementById("slogan").innerHTML = `<i>"${slogans[s]} "</i>`; s = (s+1)%slogans.length; }, 3000);

  // Particles
  const canvas = document.getElementById('particles');
  const ctx = canvas.getContext('2d');
  canvas.width = window.innerWidth; canvas.height = window.innerHeight;
  const particles = [];
  for(let i=0; i<80; i++){
    particles.push({x:Math.random()*canvas.width, y:Math.random()*canvas.height, r:Math.random()*2+1, d:Math.random()*80});
  }
  function draw(){ ctx.clearRect(0,0,canvas.width,canvas.height); ctx.fillStyle='#00ffff'; particles.forEach(p=>{ ctx.beginPath(); ctx.arc(p.x,p.y,p.r,0,Math.PI*2); ctx.fill(); p.x += (Math.random()-0.5)*2; p.y += (Math.random()-0.5)*2; if(p.x<0||p.x>canvas.width) p.x = Math.random()*canvas.width; if(p.y<0||p.y>canvas.height) p.y = Math.random()*canvas.height; }); requestAnimationFrame(draw); }
  draw();
  window.addEventListener('resize',()=>{ canvas.width = window.innerWidth; canvas.height = window.innerHeight; });
</script>
